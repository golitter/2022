### 堆积木最小操作

[P1969 [NOIP2013 提高组\] 积木大赛 - 洛谷 | 计算机科学教育新生态 (luogu.com.cn)](https://www.luogu.com.cn/problem/P1969)

```cpp
#include <bits/stdc++.h>
using namespace std;
const int N = 2e6 +21;
int a[N];
int main()
{	
	int n; cin>>n;
	for(int i = 1; i <= n ;++i) cin>>a[i];
	// for(int i = 1; i<=n; ++i) {
	// 	int x; cin>>x;
	// 	a[i] -= x;
	// }
	long long ans = 0;
	for(int i = 1; i <= n + 1; ++i) ans += a[i] > a[i - 1] ? a[i] - a[i - 1] : 0;
	cout<<ans<<endl;
}
```

### 可加可减 =(转)> 堆积木

[P7994 [USACO21DEC\] Air Cownditioning B - 洛谷 | 计算机科学教育新生态 (luogu.com.cn)](https://www.luogu.com.cn/problem/P7994)

```cpp
#include <bits/stdc++.h>
using namespace std;
const int N = 2e6 +21;
int a[N];
int main()
{	
	int n; cin>>n;
	for(int i = 1; i <= n ;++i) cin>>a[i];
	for(int i = 1; i<=n; ++i) {
		int x; cin>>x;
		a[i] -= x;
	}
	long long ans = 0;
	for(int i = 1; i <= n + 1; ++i) ans += a[i] > a[i - 1] ? a[i] - a[i - 1] : 0;
	cout<<ans<<endl;
}
```
