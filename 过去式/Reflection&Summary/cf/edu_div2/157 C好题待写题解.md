```cpp
	int x,y,k; cin>>x>>y>>k;
```

B题，求绝对值距离和最小。

排序直接出。

```cpp
void solve() {
	int n; cin>>n;
	vector<int> a(n * 2);
	for(auto &t: a) cin>>t;
	vector<array<int,2>> sg(n);
	int p = 0;
	sort(all(a));
	for(int i = 0; i < n; i ++) {
		sg[p++][0] = a[i];
	}
	p = 0;
	for(int i = n; i < 2 * n; ++i) {
		sg[p++][1] = a[i];
	}
	p = 0;
	for(int i = 1; i < n; ++i) {
		p += abs(sg[i][0] - sg[i-1][0]) + abs(sg[i][1] - sg[i-1][1]);
	}
	cout<<p<<endl;
	for(auto t: sg) cout<<t[0]<<" "<<t[1]<<endl;
}
```

C题，感觉是个很经典有贼厉害贼有水平的一道题。看看有时间写题解吗。

```cpp
void solve() {
	int n; cin>>n;
	map<int,int> mp[7];
	int cnt = 0;
	vector<string> vstr(n);
	for(auto &t: vstr) cin>>t;
	sort(all(vstr), [&](auto pre, auto suf) {
		return pre.size() < suf.size();
	});
	for(int k = 0; k < n; ++k) {
		string s = vstr[k];
		int len = s.size();
		int sum = 0, al = 0;
		for(auto &t: s) al += t - '0';
		for(int i = 0; i < len; ++i) {
			sum += s[i] - '0';
			for(int j = 1; j <= 5; ++j) {
				if(j + i + 1 != len - i - 1) continue;
				cnt += mp[j][al - 2 * sum];
			}
		}
		cnt += mp[len][al] * 2;
		sum = 0;
		for(int i = len - 1; i >= 0; --i) {
			sum += s[i] - '0';
			for(int j = 1; j <= 5; ++j) {
				int nl = len - i;
				if( j + nl != len - nl) continue;
				cnt += mp[j][al - 2 * sum];
			}
		}
		cnt++;
		mp[len][al]++;
	}
	cout<<cnt<<endl;
}
```
