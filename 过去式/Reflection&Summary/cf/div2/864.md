```cpp
void solve() {
    int n,m; cin>>n>>m;
    set<PII> s;
    s.insert({n,m}); s.insert({1,m}); s.insert({n,1}); s.insert({1,1});
    PII a,b; cin>>a.vf>>a.vs>>b.vf>>b.vs;
    int cnt = 4;
    for(auto t: s) {
        if(t == a || t == b) cnt = 2;
    }

    if(a.vf == 1 || a.vf == n || b.vf == 1 || b.vf == n) cnt  =min(cnt,3);
    if(a.vs == 1 || a.vs == m || b.vs == 1 | b.vs == m) cnt = min(cnt, 3);
    cout<<cnt<<endl;
}
```

B题，中心对称，注意细节即可。

```cpp
void solve() {
    int n,k; cin>>n>>k;
    vector<vector<int>> a(n + 1, vector<int> (n + 1));
    for(int i = 1; i <= n; ++i) {
        for(int j = 1; j <= n; ++j) a[i][j] = fread();
    }
    k <<= 1;
    for(int i = 1; i <= n; ++i) {
        for(int j = 1; j <= n; ++j) {
            if(a[i][j] != a[n + 1 - i][n + 1 - j]) k--;
        }
    }
    k >>= 1;
    puts((k < 0 || (k > 0 && k % 2 == 1 && n % 2 == 0)) ? "NO" : "YES");
}
```

C题，交互。

```cpp
void solve() {
	auto ask = [&](int x, int y) -> int {
		cout<<"? "<<x<<" "<<y<<endl;
		int t; cin>>t;
		return t;
	};
	auto inp = [&](int x, int y) -> void {
		cout<<"! "<<x<<" "<<y<<endl;
	};
	int n,m; cin>>n>>m;
	int a = ask(1,1);
	if(a >= n) {
		int b = ask(1,a + 1);
		inp(b+1, a + 1);
	} else if(a >= m) {
		int b = ask(a + 1, 1);
		inp(a + 1, b + 1);
	} else {
		int b = ask(a + 1, a + 1);
		int c = ask(a + 1, a - b + 1);
		if(c) inp(a - b + 1, a + 1);
		else inp(a + 1, a - b + 1);
	}
}
```
