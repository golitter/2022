B题，bfs板题，但是观察发现只要删除操作结束后两位数是00 25 50 75 就行。枚举这四个位置即可。

```cpp
void solve() {
    // 25 50 75 00
    string s; cin>>s;
    int n = s.size();
    int cnt = INF;
    // 00
    if(s.rfind("0") != string::npos) {
        int p2 = s.rfind("0");
        for(int i = p2-1; i >= 0; --i) {
            if(s[i] == '0') {
                cnt = min(cnt, n - i - 2);
                break;
            }
        }
    }
    // 25
    if(s.rfind("5") != string::npos) {
        int p2 = s.rfind("5");
        for(int i = p2; i >= 0; --i) {
            if(s[i] == '2') {
                cnt = min(cnt, n - i - 2);
                break;
            }
        }
    }
    // 50
    if(s.rfind("0") != string::npos) {
        int p2 = s.rfind("0");
        for(int i = p2; i >= 0; --i) {
            if(s[i] == '5') {
                cnt = min(cnt, n - i - 2);
                break;
            }
        }
    }
    // 75
    if(s.rfind("5") != string::npos) {
        int p2 = s.rfind("5");
        for(int i = p2; i >= 0; --i) {
            if(s[i] == '7') {
                cnt = min(cnt, n - i - 2);
                break;
            }
        }
    }
    if(cnt == INF) {
        cnt = n - count(all(s), '0');
    }
    cout<<cnt<<endl;
}
```

C题，贪心。一开始以为是二分呢；后面发现猫最多走n步，排序从后往前找。

```cpp
void solve() {
    int n,k; cin>>n>>k;
    vector<int> a(k);
    int sum = 0;
    for(auto &t: a) cin>>t, sum += n - t;
    sort(all(a));
    int cnt = 0;
    int al = n;
    for(int i = k-1; i >= 0; --i) {
        if(n - a[i] < al) cnt++, al -= n - a[i];
        else break;
    }
    cout<<cnt<<endl;
}
```

E，拓扑序。

```cpp
void solve() {
    int n,k; cin>>n>>k;
    vector<vector<int>> g(n + 1);
    vector<int> vis(n + 1), deg(n + 1);
    for(int i = 1; i < n; ++i) {
        int u,v; cin>>u>>v;
        g[u].push_back(v);
        g[v].push_back(u);
        deg[u]++, deg[v]++;
    }
    queue<array<int,2>> q; // idx, deg_cnt
    for(int i = 1; i <= n; ++i) if(deg[i] <= 1) q.push({i,1});
    while(q.size()) {
        auto tmp = q.front(); q.pop();
        if(vis[tmp[0]]) continue;
        vis[tmp[0]] = 1;
        if(tmp[1] >= k) continue;
        int x = tmp[0], ak = tmp[1];
        for(auto y: g[x]) {
            deg[y]--;
            if(deg[y] <= 1 && !vis[y]) q.push({y, ak + 1});
        }
    }
    int cnt = 0;
    for(int i = 1; i <= n; ++i) cnt += !vis[i];
    cout<<cnt<<endl;
}
```
