C题，找满足：
$$
l \le a + b \le r \\
gcd(a,b) \neq 1
$$
的a和b。

思路：如果要满足`gcd(a,b) != 1`，那么一定存在一个`c`，使`gcd(a,b) = c`，且`a + b = (k + 1) * c (c > 1)`，对于限制条件一，只需要让`l <= a + b <= r`，即`l <= (k + 1) *c <= r`。枚举c，找满足条件的k就行，同时也可以对k的值进行优化，将第一个遍历的k值就满足`l <= (k + 1) * c`，遍历到`(k + 1) * c <= r`。

代码如下：

```cpp
void solve() {
    int l,r; cin>>l>>r; // l 和 r
    int lc = sqrt(r); // 减少枚举c的个数，大于sqrt(r)之后一定不满足 (k + 1) * c <= r 这一个条件
    for(int c = 2; c <= lc; ++c) {
        // 开始遍历的k要大于 l/c 向上取整
        int st = l / c + (l % c == 0 ? 0 : 1); 
        int ed = r / c;
        // 这里将上述的 k + 1 转为 k，毕竟都一样
        for(int k = max(2,st); k <= ed; ++k) {
            // 满足条件输出即可。
            cout<<(k - 1) * c<<" "<<c<<endl;
            return ;
        }
    }
    cout<<-1<<endl;
}
```

D题，容斥。排列是自定义的，要满足这个条件的最大值就是找到对于x，1到n之内有多少个倍数，对于y，找1到n之内有多少个倍数，对x和y的个数还有删除x和y共有的倍数。之后，对x和y剩余的倍数个数进行两个等差数列。x，从最大值n开始减；，y，从最小值开始增。

```cpp
void solve() {
    LL n,x,y; cin>>n>>x>>y;
    // x和y出现的次数
    LL cx = n / x, cy = n / y;
    // 找x和y的最大公倍数
    LL lcm = x * y / __gcd(x,y);
    // 1到n之内有多少个最大公倍数的倍数个数
    LL xy = n / (lcm);

    // x和y出现的朴素次数减去 最大公倍数的倍数个数（即共有的倍数
    cx -= xy;
    cy -= xy;
    LL ans = 0;
    // 对x，n n-1 n-2 n-3 n-4 ... 1
    ans += cx * n + ( cx * (cx - 1) * (-1LL)) / 2;
    // 对y, 1 2 ... n-2 n-1 n
    ans -= cy * (cy + 1) / 2;
    cout<<ans<<endl;
}
```

F题，基环树。对于每一个环，删除最小点权的，断环成链，之后拓扑序。

[基环树瞎吹 - QuantAsk的博客 - 洛谷博客 (luogu.com.cn)](https://www.luogu.com.cn/blog/user52918/qian-tan-ji-huan-shu)

```cpp
void solve() {
    int n; cin>>n;
    vector<int> a(n + 1), b(n + 1), vis(n + 1), in(n + 1),cycle,rcd(n + 1),nin(n + 1);
    vector<vector<PII>> g(n + 1); // 记录 <u,v> (id) 这个id是防止 <i,a[i]> <a[i],i>
    vector<vector<int>> newg(n + 1); // 断环成链的新边
    for(int i = 1; i <= n; ++i) {
        cin>>a[i];
        // <i,a[i]>(i) <a[i],i>(i)
        g[i].push_back({a[i], i});
        g[a[i]].push_back({i,i});
    }
    for(int i = 1; i <= n; ++i) cin>>b[i];

    // 找树内环
    auto dfs = [&](auto &&dfs, int u, int id) -> void {
        vis[u] = 1;
        // 出度（当作入度也行，就是记录是否dfs里有这个点
        in[u] = 1;
        for(auto t: g[u]) {
            // <u,y>(id)
            int y = t.vf, tid = t.vs;
            // id不能跟上一个边相同
            if(tid == id) continue;
            // 如果y没有在dfs路径中，未成环
            if(!vis[y]) dfs(dfs,y,tid);
            // 相遇，成环
            else if(in[y]) {
                int cy = y;
                // 将环上下标输入到cycle中
                cycle.push_back(cy);
                while(1) {
                    cy = a[cy];
                    if(cy == y) break;
                    cycle.push_back(cy);
                }
            }
        }
        // u不在环中，（回溯
        in[u] = 0;
    };
    for(int i = 1; i <= n; ++i) {
        // 如果为1，表示之前在成环路径中
        if(vis[i]) continue;
        cycle.clear(); // 清空
        dfs(dfs,i,-1);
        int t = -1;
        // 找最小的b值
        for(auto y: cycle) {
            if(t == -1 || b[y] < b[t]) t = y;
        }
        // 将这个环中，最小b值的下标进行标记
        rcd[t] = 1;
    }
    for(int i = 1; i <= n; ++i) {
        // 不在环中，正常树结构
        if(!rcd[i]) {
            newg[i].push_back(a[i]);
            // 记录这个位置的入度
            nin[a[i]]++;
        }
    }
    queue<int> q;
    for(int i = 1; i <= n; ++i) {
        // 入度为0，是拓扑序起始位置，push进队列
        if(!nin[i]) q.push(i);
    }
    // 拓扑序
    while(q.size()) {
        int t = q.front(); q.pop();
        cout<<t<<" ";
        for(auto y: newg[t]) {
            nin[y]--;
            if(!nin[y]) {
                q.push(y);
            }
        }
    }
    puts("");

}
```
