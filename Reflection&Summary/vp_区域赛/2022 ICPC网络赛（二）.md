void write(__int128_t value) {
    if(value > 9) {
        write(value / 10);
    }
    putchar( (value % 10) + '0');
}

void solve() {
    LL k,x,y; cin>>k>>x>>y;
    __int128_t psx = 1, psy = 1;

    // 向上取整
    auto up = [&](LL a, LL b) -> LL {
        // return a / b + (a % b == 0 ? 0 : 1);
        return (a + b - 1) / b;
    };
    // 前一秒有多少个节点
    auto presecond = [&](LL x) -> __int128_t {
        __int128_t tmp = 1;
        while(tmp < x) tmp *= (k + 1);
        tmp /= (k + 1);
        return tmp;
    };
    psx = presecond(x);
    psy = presecond(y);
    while(x != y) {
        if(x < y) swap(x,y), swap(psx, psy);
        x = up(x - psx, k);
        psx = presecond(x);
    }
    cout<<x<<endl;
}
