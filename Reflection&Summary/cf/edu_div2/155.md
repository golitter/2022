void solve() {
    int n; cin>>n;
    vector<LL> a(n + 1);
    for(int i = 1; i <= n; ++i) cin>>a[i], a[i] ^= a[i-1];
    int ans = 0;
    for(int i = 0; i <= 30; ++i) {
        vector<LL> cnt({1,0}), sum({0,0});
        for(int j = 1; j <= n; ++j) {
            int x = a[j] >> i & 1;
            ans += (j * cnt[!x] - sum[!x]) % mod * (1 << i) % mod;
            ans %= mod;
            cnt[x]++;
            sum[x] += j;
        }
    }
    cout<<ans<<endl;
}
