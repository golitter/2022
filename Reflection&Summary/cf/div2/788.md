A题，读懂题就行，一开始没仔细读题，以为是正负数进行交换呢，模拟样例没过，之后又看了看，发现是传递负号。那就很简单了。

```cpp
void solve() {
    int n; cin>>n;
    vector<int> a(n);
    int neg = 0;
    for(auto &t: a) cin>>t, neg += t < 0, t = abs(t);
    for(auto &t: a) {
        if(neg <= 0) break;
        t =  -t;
        neg--;
    }
    bool ok = true;
    for(int i = 1; i < n; ++i) ok &= a[i] >= a[i-1];
    puts(ok ? "YES" : "NO");
}
```
