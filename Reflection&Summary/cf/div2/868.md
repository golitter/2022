```cpp
void solve() {
    int n,k; cin>>n>>k;
    vector<int> a(n + 1);
    for(int i = 0; i <= n; ++i) {
        int len = i;
        for(int j = 1; j <= len; ++j) a[j] = 1;
        for(int j = len + 1; j <= n; ++j) a[j] = -1;
        int cnt = 0;
        for(int k = 1; k <= n; ++k) {
            for(int z = k + 1; z <= n; ++z) {
                cnt += a[k] * a[z] == 1;
            }
        }
        if(cnt == k) {
            puts("YES");
            for(int k = 1; k <= n; ++k) cout<<a[k]<<" \n"[k == n];
            return ;
        }
    }
    puts("NO");
}
```

```cpp
    }
    set<int>s;
    for(int i = 1; i <= k; ++i) {
        for(int j = i; j <= n; j += k) {
            if(mp[i].find(j) == mp[i].end()) s.insert(j);
        }

}
```
